# Hero Flowing Mesh — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Einen subtilen animierten Flowing-Mesh-Hintergrundeffekt in die Hero-Section der Startseite integrieren, der auf den Cursor reagiert.

**Architecture:** Eine neue `HeroMesh.astro`-Komponente enthält vier animierte Orbs in einem `blur()`-Container. Die autonome CSS-Animation läuft dauerhaft; ein JS-Lerp-Loop verschiebt den gesamten Container träge in Richtung Cursor. Zwei neue OKLCH-Farbtokens in `globals.css` liefern die Mesh-Farben.

**Tech Stack:** Astro, CSS Custom Properties, CSS `@keyframes`, `requestAnimationFrame`, `pointermove`

### Task 2: HeroMesh.astro erstellen

**Files:**
- Create: `src/components/astro/HeroMesh.astro`

- [ ] **Schritt 1: Datei erstellen**

  `src/components/astro/HeroMesh.astro` mit folgendem Inhalt anlegen:

  ```astro
  ---
  ---

  <div class="mesh-container" aria-hidden="true">
    <div class="mesh-orb orb-1"></div>
    <div class="mesh-orb orb-2"></div>
    <div class="mesh-orb orb-3"></div>
    <div class="mesh-orb orb-4"></div>
  </div>

  <style>
    .mesh-container {
      position: absolute;
      inset: -12%;
      width: 124%;
      height: 124%;
      z-index: 0;
      pointer-events: none;
      filter: blur(40px);
      opacity: 0.38;
      will-change: transform;
    }

    .mesh-orb {
      position: absolute;
      border-radius: 50%;
    }

    /* Orb 1 — warm, oben links */
    .orb-1 {
      width: 55%;
      height: 45%;
      top: 20%;
      left: 15%;
      background: var(--mesh-warm);
      animation: orb-drift-1 18s ease-in-out infinite alternate;
    }

    /* Orb 2 — cool, oben rechts */
    .orb-2 {
      width: 45%;
      height: 55%;
      top: 8%;
      right: 8%;
      background: var(--mesh-cool);
      animation: orb-drift-2 22s ease-in-out infinite alternate;
      animation-delay: -5s;
    }

    /* Orb 3 — warm, unten rechts */
    .orb-3 {
      width: 60%;
      height: 40%;
      bottom: 8%;
      left: 45%;
      background: var(--mesh-warm);
      animation: orb-drift-3 16s ease-in-out infinite alternate;
      animation-delay: -8s;
    }

    /* Orb 4 — cool, unten links */
    .orb-4 {
      width: 40%;
      height: 60%;
      bottom: 12%;
      left: 18%;
      background: var(--mesh-cool);
      animation: orb-drift-4 20s ease-in-out infinite alternate;
      animation-delay: -12s;
    }

    @keyframes orb-drift-1 {
      0%   { transform: translate(0%, 0%)   scale(1)    rotate(0deg);  }
      33%  { transform: translate(8%, -6%)  scale(1.05) rotate(2deg);  }
      66%  { transform: translate(-5%, 8%)  scale(0.96) rotate(-1deg); }
      100% { transform: translate(6%, 5%)   scale(1.03) rotate(3deg);  }
    }

    @keyframes orb-drift-2 {
      0%   { transform: translate(0%, 0%)   scale(1)    rotate(0deg);  }
      40%  { transform: translate(-9%, 5%)  scale(1.04) rotate(-3deg); }
      80%  { transform: translate(5%, -8%)  scale(0.97) rotate(2deg);  }
      100% { transform: translate(-4%, 6%)  scale(1.02) rotate(-2deg); }
    }

    @keyframes orb-drift-3 {
      0%   { transform: translate(0%, 0%)   scale(1)    rotate(0deg);  }
      35%  { transform: translate(-7%, -4%) scale(1.06) rotate(1deg);  }
      70%  { transform: translate(6%, 7%)   scale(0.95) rotate(-3deg); }
      100% { transform: translate(-3%, -6%) scale(1.04) rotate(2deg);  }
    }

    @keyframes orb-drift-4 {
      0%   { transform: translate(0%, 0%)   scale(1)    rotate(0deg);  }
      45%  { transform: translate(7%, -5%)  scale(1.03) rotate(-2deg); }
      75%  { transform: translate(-6%, 6%)  scale(0.98) rotate(3deg);  }
      100% { transform: translate(5%, -4%)  scale(1.05) rotate(-1deg); }
    }

    @media (prefers-reduced-motion: reduce) {
      .mesh-orb {
        animation: none;
      }
      .mesh-container {
        opacity: 0.15;
      }
    }
  </style>

  <script>
    (function () {
      if (window.matchMedia('(prefers-reduced-motion: reduce)').matches) return;
      if (!window.matchMedia('(pointer: fine)').matches) return;

      const container = document.querySelector<HTMLElement>('.mesh-container');
      const hero = container?.closest<HTMLElement>('section');
      if (!container || !hero) return;

      const MAX_X = 6;
      const MAX_Y = 4;
      const LERP  = 0.035;

      let targetX = 0, targetY = 0;
      let currentX = 0, currentY = 0;
      let rafId: number;

      hero.addEventListener('pointermove', (e: PointerEvent) => {
        const rect = hero.getBoundingClientRect();
        targetX = ((e.clientX - rect.left) / rect.width  - 0.5) * MAX_X * 2;
        targetY = ((e.clientY - rect.top)  / rect.height - 0.5) * MAX_Y * 2;
      });

      hero.addEventListener('pointerleave', () => {
        targetX = 0;
        targetY = 0;
      });

      function tick(): void {
        currentX += (targetX - currentX) * LERP;
        currentY += (targetY - currentY) * LERP;
        container!.style.transform =
          `translate(${currentX.toFixed(3)}%, ${currentY.toFixed(3)}%)`;
        rafId = requestAnimationFrame(tick);
      }

      tick();

      // Cleanup on Astro View Transition navigation
      document.addEventListener('astro:before-swap', () => {
        cancelAnimationFrame(rafId);
      });
    })();
  </script>
  ```
